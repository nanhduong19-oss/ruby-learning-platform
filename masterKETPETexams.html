// ================================================================ // Ruby School – Reading Practice · Google Apps Script v2 // Deploy as: Web App · Execute as: Me · Access: Anyone // ================================================================ const SHEET_ID = '1Jl2AXS49Qg3arlexUR3r_V5sPkiqRg2aCSuuDk0laAo'; // ── Practice tabs ── const PRACTICE_TABS = [ 'KET_Part1','KET_Part2','KET_Part3','KET_Part4','KET_Part5', 'PET_Part1','PET_Part2','PET_Part3','PET_Part4','PET_Part5','PET_Part6' ]; const PRACTICE_HEADERS = [ 'Timestamp','Duration','StudentID','Name','Class', 'Level','Part','TotalQ','Correct','Wrong','Percentage' ]; // ── Full Test tabs ── const KET_FT_HEADERS = [ 'Timestamp','Duration','StudentID','Name','Class', 'Total Correct','Total Wrong','Percentage','Grade','Cambridge Score','Violations', 'P1 Correct','P1 Source', 'P2 Correct','P2 Source', 'P3 Correct','P3 Source', 'P4 Correct','P4 Source', 'P5 Correct','P5 Source' ]; const PET_FT_HEADERS = [ 'Timestamp','Duration','StudentID','Name','Class', 'Total Correct','Total Wrong','Percentage','Grade','Cambridge Score','Violations', 'P1 Correct','P1 Source', 'P2 Correct','P2 Source', 'P3 Correct','P3 Source', 'P4 Correct','P4 Source', 'P5 Correct','P5 Source', 'P6 Correct','P6 Source' ]; // ── Wrong Answers tab ── const WRONG_TAB = 'WrongAnswers'; const WRONG_HEADERS = [ 'Timestamp','StudentID','PartKey','QuestionIDs' ]; // ================================================================ // MAIN ENTRY POINTS // ================================================================ function doPost(e) { try { let payload; if (e.parameter && e.parameter.payload) { payload = JSON.parse(e.parameter.payload); } else if (e.postData && e.postData.contents) { payload = JSON.parse(e.postData.contents); } else { throw new Error('No payload received'); } const action = payload.action; if (action === 'saveSession') return saveSession(payload); if (action === 'saveFullTest') return saveFullTest(payload); return jsonResp({ ok: false, error: 'Unknown action: ' + action }); } catch (err) { return jsonResp({ ok: false, error: err.message }); } } function doGet(e) { const action = (e && e.parameter && e.parameter.action) || ''; if (action === 'getWrongAnswers') { const studentId = e.parameter.studentId || ''; const partKey = e.parameter.partKey || ''; if (!studentId || !partKey) { return jsonResp({ ok: false, questionIds: [], error: 'Missing studentId or partKey' }); } return getWrongAnswers(studentId, partKey); } return jsonResp({ ok: true, message: 'Ruby School Apps Script v2 is running.' }); } // ================================================================ // SAVE SESSION (Practice mode) // ================================================================ function saveSession(p) { const lock = LockService.getDocumentLock(); try { lock.waitLock(15000); } catch (lockErr) { return jsonResp({ ok: false, error: 'Server busy - please try again.' }); } try { const ss = SpreadsheetApp.openById(SHEET_ID); const tab = ensurePracticeTab(ss, p.tab); const ts = vnTimestamp(); tab.appendRow([ ts, p.duration || '', p.studentId || '', p.studentName || '', p.class || '', p.level || '', p.part || '', p.totalQ || 0, p.correct || 0, p.wrong || 0, (p.percentage || 0) + '%' ]); SpreadsheetApp.flush(); if (tab.getLastRow() <= 3) tab.autoResizeColumns(1, PRACTICE_HEADERS.length); // ── Save wrong answers for priority practice ── var wrongIds = p.wrongIds; var partKey = p.partKey; if (wrongIds && wrongIds.length > 0 && partKey && p.studentId) { saveWrongAnswers(ss, p.studentId, partKey, wrongIds); } return jsonResp({ ok: true, message: 'Session saved.' }); } catch (err) { return jsonResp({ ok: false, error: err.message }); } finally { lock.releaseLock(); } } // ================================================================ // SAVE FULL TEST // ================================================================ function saveFullTest(p) { const lock = LockService.getDocumentLock(); try { lock.waitLock(15000); } catch (lockErr) { return jsonResp({ ok: false, error: 'Server busy - please try again.' }); } try { const ss = SpreadsheetApp.openById(SHEET_ID); const level = (p.level || '').toLowerCase(); const isKET = level === 'ket'; const tabName = isKET ? 'KET_FullTest' : 'PET_FullTest'; const headers = isKET ? KET_FT_HEADERS : PET_FT_HEADERS; const tab = ensureFullTestTab(ss, tabName, headers); const ts = vnTimestamp(); // Build part columns from payload.parts array // KET: 5 parts (P1-P5), PET: 6 parts (P1-P6) var parts = p.parts || []; var partCols = []; var partCount = isKET ? 5 : 6; for (var i = 0; i < partCount; i++) { var pt = parts[i] || {}; partCols.push(pt.correct || 0); partCols.push(pt.source || '-'); } var row = [ ts, p.duration || '', p.studentId || '', p.name || '', p.class || '', p.totalCorrect || 0, p.totalWrong || 0, (p.percentage || 0) + '%', p.grade || '', p.score || 0, p.violations || 0 ].concat(partCols); tab.appendRow(row); SpreadsheetApp.flush(); if (tab.getLastRow() <= 3) tab.autoResizeColumns(1, headers.length); // ── Save wrong answers from each part ── if (p.studentId) { for (var j = 0; j < parts.length; j++) { var pt2 = parts[j]; if (pt2.wrongQids && pt2.wrongQids.length > 0) { // Derive partKey from label: "P1 Signs" -> index 0 -> ket_part1 / pet_part1 var partKey2 = derivePartKey(level, j); if (partKey2) { saveWrongAnswers(ss, p.studentId, partKey2, pt2.wrongQids); } } } } return jsonResp({ ok: true, message: 'Full test saved.' }); } catch (err) { return jsonResp({ ok: false, error: err.message }); } finally { lock.releaseLock(); } } // ================================================================ // GET WRONG ANSWERS (for priority practice) // ================================================================ function getWrongAnswers(studentId, partKey) { try { const ss = SpreadsheetApp.openById(SHEET_ID); const sheet = ss.getSheetByName(WRONG_TAB); if (!sheet || sheet.getLastRow() <= 1) { return jsonResp({ ok: true, questionIds: [] }); } var data = sheet.getDataRange().getValues(); // Find the LATEST row for this student + partKey // Columns: Timestamp(0), StudentID(1), PartKey(2), QuestionIDs(3) var latestIds = []; for (var i = data.length - 1; i >= 1; i--) { if (String(data[i][1]).toUpperCase() === studentId.toUpperCase() && String(data[i][2]) === partKey) { var raw = String(data[i][3]); if (raw) latestIds = raw.split(',').map(function(s) { return s.trim(); }).filter(Boolean); break; // only need the latest entry } } return jsonResp({ ok: true, questionIds: latestIds }); } catch (err) { return jsonResp({ ok: false, questionIds: [], error: err.message }); } } // ================================================================ // HELPERS // ================================================================ // ── Save wrong answer IDs to WrongAnswers tab ── function saveWrongAnswers(ss, studentId, partKey, wrongIds) { var sheet = ss.getSheetByName(WRONG_TAB); if (!sheet) { sheet = ss.insertSheet(WRONG_TAB); sheet.appendRow(WRONG_HEADERS); styleHeader(sheet, WRONG_HEADERS.length); } else if (sheet.getLastRow() === 0) { sheet.appendRow(WRONG_HEADERS); styleHeader(sheet, WRONG_HEADERS.length); } var ts = vnTimestamp(); // Store as comma-separated string var idsStr = wrongIds.join(', '); sheet.appendRow([ts, studentId, partKey, idsStr]); } // ── Derive partKey from level + part index ── function derivePartKey(level, idx) { var ketParts = ['ket_part1','ket_part2','ket_part3','ket_part4','ket_part5']; var petParts = ['pet_part1','pet_part2','pet_part3','pet_part4','pet_part5','pet_part6']; var arr = level === 'ket' ? ketParts : petParts; return arr[idx] || null; } // ── Ensure Practice tab exists ── function ensurePracticeTab(ss, tabName) { if (!PRACTICE_TABS.includes(tabName)) throw new Error('Invalid tab: ' + tabName); var sheet = ss.getSheetByName(tabName); if (!sheet) { sheet = ss.insertSheet(tabName); sheet.appendRow(PRACTICE_HEADERS); styleHeader(sheet, PRACTICE_HEADERS.length); } else if (sheet.getLastRow() === 0) { sheet.appendRow(PRACTICE_HEADERS); styleHeader(sheet, PRACTICE_HEADERS.length); } return sheet; } // ── Ensure Full Test tab exists ── function ensureFullTestTab(ss, tabName, headers) { var sheet = ss.getSheetByName(tabName); if (!sheet) { sheet = ss.insertSheet(tabName); sheet.appendRow(headers); styleHeader(sheet, headers.length); } else if (sheet.getLastRow() === 0) { sheet.appendRow(headers); styleHeader(sheet, headers.length); } return sheet; } // ── Style header row ── function styleHeader(sheet, colCount) { var h = sheet.getRange(1, 1, 1, colCount); h.setFontWeight('bold') .setBackground('#b91c1c') .setFontColor('#ffffff') .setHorizontalAlignment('center'); sheet.setFrozenRows(1); } // ── Vietnam timestamp (UTC+7) ── function vnTimestamp() { var now = new Date(); var vn = new Date(now.getTime() + 7 * 60 * 60 * 1000); return Utilities.formatDate(vn, 'UTC', 'dd/MM/yy HH:mm:ss'); } // ── JSON response ── function jsonResp(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
